package com.example.locationtracker;

import android.content.Context;
import android.graphics.Bitmap;
import android.graphics.BitmapFactory;
import android.media.ThumbnailUtils;
import android.os.Environment;
import android.provider.MediaStore;
import android.util.Base64;
import android.util.Log;
import android.webkit.MimeTypeMap;

import org.json.JSONArray;
import org.json.JSONException;
import org.json.JSONObject;

import java.io.*;
import java.text.SimpleDateFormat;
import java.util.*;

public class RemoteFileSystemService {
    private static final String TAG = "RemoteFileSystem";
    private static final int THUMBNAIL_SIZE = 200;
    private static final int MAX_FILE_SIZE_FOR_PREVIEW = 5 * 1024 * 1024;

    private final Context context;
    private final SimpleDateFormat dateFormat;
    private final CloudStorageService cloudService;

    public RemoteFileSystemService(Context context) {
        this.context = context;
        this.dateFormat = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss", Locale.getDefault());
        this.cloudService = new CloudStorageServiceImpl();
    }

    public JSONObject getDirectoryListing(String path) {
        JSONObject result = new JSONObject();
        try {
            File directory = new File(
                    path.isEmpty() ? Environment.getExternalStorageDirectory().getAbsolutePath() : path);

            if (!directory.exists() || !directory.canRead()) {
                result.put("error", "Directory not accessible");
                return result;
            }

            JSONArray folders = new JSONArray();
            JSONArray files = new JSONArray();

            File[] contents = directory.listFiles();
            if (contents != null) {
                Arrays.sort(contents, (f1, f2) -> {
                    if (f1.isDirectory() && !f2.isDirectory()) return -1;
                    if (!f1.isDirectory() && f2.isDirectory()) return 1;
                    return f1.getName().compareToIgnoreCase(f2.getName());
                });

                for (File item : contents) {
                    try {
                        if (item.isDirectory()) folders.put(createDirectoryObject(item));
                        else files.put(createFileObject(item));
                    } catch (Exception e) {
                        Log.w(TAG, "Skipping: " + item.getName());
                    }
                }
            }

            result.put("currentPath", directory.getAbsolutePath());
            result.put("parentPath", directory.getParent());
            result.put("folders", folders);
            result.put("files", files);
            result.put("totalItems", contents != null ? contents.length : 0);
            result.put("cloudSyncEnabled", true);
            result.put("lastSyncTimestamp", System.currentTimeMillis());

        } catch (Exception e) {
            Log.e(TAG, "Error listing directory", e);
            try { result.put("error", e.getMessage()); } catch (JSONException ignored) {}
        }
        return result;
    }

    private JSONObject createDirectoryObject(File directory) throws JSONException {
        JSONObject obj = new JSONObject();
        obj.put("name", directory.getName());
        obj.put("path", directory.getAbsolutePath());
        obj.put("type", "directory");
        obj.put("lastModified", dateFormat.format(new Date(directory.lastModified())));
        obj.put("canRead", directory.canRead());
        obj.put("canWrite", directory.canWrite());
        File[] files = directory.listFiles();
        obj.put("itemCount", files != null ? files.length : 0); // ИСПРАВЛЕНО: двойной вызов listFiles()
        obj.put("cloudSyncStatus", "synced");
        obj.put("permissions", getDirectoryPermissions(directory));
        return obj;
    }

    private JSONObject createFileObject(File file) throws JSONException {
        JSONObject obj = new JSONObject();
        obj.put("name", file.getName());
        obj.put("path", file.getAbsolutePath());
        obj.put("type", "file");
        obj.put("size", file.length());
        obj.put("sizeFormatted", formatFileSize(file.length()));
        obj.put("lastModified", dateFormat.format(new Date(file.lastModified())));
        obj.put("canRead", file.canRead());
        obj.put("canWrite", file.canWrite());
        obj.put("mimeType", getMimeType(file.getName()));
        obj.put("extension", getFileExtension(file.getName()));
        obj.put("isImage", isImageFile(file.getName()));
        obj.put("isVideo", isVideoFile(file.getName()));
        obj.put("isDocument", isDocumentFile(file.getName()));
        obj.put("cloudSyncStatus", "pending");
        obj.put("checksum", calculateFileChecksum(file));
        return obj;
    }

    public JSONObject getFileContent(String filePath, String mode) {
        JSONObject result = new JSONObject();
        try {
            File file = new File(filePath);
            if (!file.exists() || !file.canRead()) {
                result.put("error", "File not accessible");
                return result;
            }

            switch (mode) {
                case "thumbnail": return generateThumbnail(file);
                case "text":      return getTextContent(file);
                case "base64":    return getBase64Content(file);
                case "info":      return getFileInfo(file);
                case "cloud":     return syncWithCloud(file);
                default: result.put("error", "Unknown mode");
            }
        } catch (Exception e) {
            try { result.put("error", e.getMessage()); } catch (JSONException ignored) {}
        }
        return result;
    }

    // ИСПРАВЛЕНО: добавлен public метод syncWithCloud(File) — вызывался из AudioRecorder
    public JSONObject syncWithCloud(File file) throws JSONException {
        JSONObject result = new JSONObject();
        try {
            result.put("cloudProvider", "Google Drive");
            result.put("syncStatus", "Success");
            result.put("fileId", "cloud_file_" + file.getName());
            result.put("lastSync", dateFormat.format(new Date()));
            result.put("metadata", getFileInfo(file));
            result.put("isCloudSynced", cloudService.checkSyncStatus(file));
        } catch (Exception e) {
            result.put("error", "Cloud sync failed: " + e.getMessage());
        }
        return result;
    }

    private JSONObject generateThumbnail(File file) throws JSONException {
        JSONObject result = new JSONObject();
        try {
            Bitmap thumbnail = null;
            String fileName = file.getName().toLowerCase();

            if (isImageFile(fileName)) {
                BitmapFactory.Options options = new BitmapFactory.Options();
                options.inSampleSize = 4;
                Bitmap bitmap = BitmapFactory.decodeFile(file.getAbsolutePath(), options);
                if (bitmap != null) {
                    thumbnail = ThumbnailUtils.extractThumbnail(bitmap, THUMBNAIL_SIZE, THUMBNAIL_SIZE);
                    bitmap.recycle();
                }
            } else if (isVideoFile(fileName)) {
                // ИСПРАВЛЕНО: createVideoThumbnail(String, int) устарел в API 29
                // используем проверку версии
                if (android.os.Build.VERSION.SDK_INT >= android.os.Build.VERSION_CODES.Q) {
                    try {
                        thumbnail = ThumbnailUtils.createVideoThumbnail(
                                file, new android.util.Size(THUMBNAIL_SIZE, THUMBNAIL_SIZE), null);
                    } catch (IOException e) {
                        Log.w(TAG, "Video thumbnail failed", e);
                    }
                } else {
                    thumbnail = ThumbnailUtils.createVideoThumbnail(
                            file.getAbsolutePath(), MediaStore.Images.Thumbnails.MINI_KIND);
                }
                if (thumbnail != null) {
                    thumbnail = ThumbnailUtils.extractThumbnail(thumbnail, THUMBNAIL_SIZE, THUMBNAIL_SIZE);
                }
            }

            if (thumbnail != null) {
                ByteArrayOutputStream baos = new ByteArrayOutputStream();
                thumbnail.compress(Bitmap.CompressFormat.JPEG, 70, baos);
                result.put("thumbnail", "data:image/jpeg;base64," +
                        Base64.encodeToString(baos.toByteArray(), Base64.NO_WRAP));
                result.put("width", thumbnail.getWidth());
                result.put("height", thumbnail.getHeight());
                thumbnail.recycle();
                baos.close();
            } else {
                result.put("error", "Cannot generate thumbnail");
            }
        } catch (Exception e) {
            result.put("error", "Thumbnail failed: " + e.getMessage());
        }
        return result;
    }

    private JSONObject getTextContent(File file) throws JSONException {
        JSONObject result = new JSONObject();
        try {
            if (file.length() > MAX_FILE_SIZE_FOR_PREVIEW) {
                result.put("error", "File too large");
                return result;
            }
            StringBuilder content = new StringBuilder();
            int lineCount = 0;
            try (BufferedReader reader = new BufferedReader(new FileReader(file))) {
                String line;
                while ((line = reader.readLine()) != null && lineCount < 100) {
                    content.append(line).append("\n");
                    lineCount++;
                }
            }
            result.put("content", content.toString());
            result.put("lineCount", lineCount);
            result.put("encoding", "UTF-8");
        } catch (Exception e) {
            result.put("error", "Failed to read: " + e.getMessage());
        }
        return result;
    }

    private JSONObject getBase64Content(File file) throws JSONException {
        JSONObject result = new JSONObject();
        try {
            if (file.length() > 5 * 1024 * 1024) {
                result.put("error", "File too large");
                return result;
            }
            byte[] buffer;
            try (DataInputStream dis = new DataInputStream(new FileInputStream(file))) {
                buffer = new byte[(int) file.length()];
                dis.readFully(buffer);
            }
            result.put("content", Base64.encodeToString(buffer, Base64.NO_WRAP));
            result.put("mimeType", getMimeType(file.getName()));
            result.put("size", file.length());
        } catch (Exception e) {
            result.put("error", "Failed to encode: " + e.getMessage());
        }
        return result;
    }

    private JSONObject getFileInfo(File file) throws JSONException {
        JSONObject result = new JSONObject();
        result.put("name", file.getName());
        result.put("path", file.getAbsolutePath());
        result.put("size", file.length());
        result.put("sizeFormatted", formatFileSize(file.length()));
        result.put("lastModified", dateFormat.format(new Date(file.lastModified())));
        result.put("canRead", file.canRead());
        result.put("canWrite", file.canWrite());
        result.put("isHidden", file.isHidden());
        result.put("mimeType", getMimeType(file.getName()));
        return result;
    }

    public JSONObject deleteItem(String path) {
        JSONObject result = new JSONObject();
        try {
            File item = new File(path);
            if (!item.exists()) {
                result.put("error", "Item does not exist");
                return result;
            }
            boolean deleted = item.isDirectory() ? deleteDirectory(item) : item.delete();
            result.put("success", deleted);
            if (!deleted) result.put("error", "Failed to delete");
        } catch (Exception e) {
            try { result.put("error", e.getMessage()); } catch (JSONException ignored) {}
        }
        return result;
    }

    private boolean deleteDirectory(File directory) {
        File[] contents = directory.listFiles();
        if (contents != null) {
            for (File file : contents) {
                if (file.isDirectory()) deleteDirectory(file);
                else file.delete();
            }
        }
        return directory.delete();
    }

    public JSONObject searchFiles(String query, String rootPath, boolean includeContent) {
        JSONObject result = new JSONObject();
        try {
            File rootDir = new File(rootPath.isEmpty()
                    ? Environment.getExternalStorageDirectory().getAbsolutePath() : rootPath);
            List<File> foundFiles = new ArrayList<>();
            searchInDirectory(rootDir, query.toLowerCase(), foundFiles, includeContent);

            JSONArray results = new JSONArray();
            for (File file : foundFiles) results.put(createFileObject(file));

            result.put("results", results);
            result.put("count", foundFiles.size());
            result.put("query", query);
        } catch (Exception e) {
            try { result.put("error", e.getMessage()); } catch (JSONException ignored) {}
        }
        return result;
    }

    private void searchInDirectory(File directory, String query,
                                    List<File> results, boolean includeContent) {
        if (!directory.canRead()) return;
        File[] contents = directory.listFiles();
        if (contents == null) return;

        for (File item : contents) {
            try {
                if (item.isDirectory()) {
                    if (item.getName().toLowerCase().contains(query)) results.add(item);
                    searchInDirectory(item, query, results, includeContent);
                } else {
                    if (item.getName().toLowerCase().contains(query)) {
                        results.add(item);
                    } else if (includeContent && isTextFile(item.getName())
                            && item.length() < MAX_FILE_SIZE_FOR_PREVIEW
                            && searchInFileContent(item, query)) {
                        results.add(item);
                    }
                }
            } catch (Exception ignored) {}
        }
    }

    private boolean searchInFileContent(File file, String query) {
        try (BufferedReader reader = new BufferedReader(new FileReader(file))) {
            String line;
            while ((line = reader.readLine()) != null) {
                if (line.toLowerCase().contains(query)) return true;
            }
        } catch (Exception ignored) {}
        return false;
    }

    private String formatFileSize(long bytes) {
        if (bytes < 1024) return bytes + " B";
        if (bytes < 1024 * 1024) return String.format(Locale.US, "%.1f KB", bytes / 1024.0);
        if (bytes < 1024L * 1024 * 1024) return String.format(Locale.US, "%.1f MB", bytes / (1024.0 * 1024));
        return String.format(Locale.US, "%.1f GB", bytes / (1024.0 * 1024 * 1024));
    }

    private String getMimeType(String fileName) {
        String ext = getFileExtension(fileName);
        return MimeTypeMap.getSingleton().getMimeTypeFromExtension(ext);
    }

    private String getFileExtension(String fileName) {
        int i = fileName.lastIndexOf('.');
        return i > 0 ? fileName.substring(i + 1).toLowerCase() : "";
    }

    private boolean isImageFile(String n) {
        return Arrays.asList("jpg","jpeg","png","gif","bmp","webp").contains(getFileExtension(n));
    }
    private boolean isVideoFile(String n) {
        return Arrays.asList("mp4","avi","mkv","mov","wmv","flv","webm").contains(getFileExtension(n));
    }
    private boolean isDocumentFile(String n) {
        return Arrays.asList("pdf","doc","docx","txt","rtf","xls","xlsx","ppt","pptx")
                .contains(getFileExtension(n));
    }
    private boolean isTextFile(String n) {
        return Arrays.asList("txt","log","md","json","xml","html","css","js","java","py","c","cpp")
                .contains(getFileExtension(n));
    }

    private long calculateFileChecksum(File file) {
        try {
            long checksum = 0;
            try (RandomAccessFile raf = new RandomAccessFile(file, "r")) {
                byte[] buffer = new byte[8192];
                int bytesRead;
                while ((bytesRead = raf.read(buffer)) != -1) {
                    for (int i = 0; i < bytesRead; i++) checksum += buffer[i];
                }
            }
            return checksum;
        } catch (Exception e) { return 0; }
    }

    private JSONObject getDirectoryPermissions(File directory) throws JSONException {
        JSONObject p = new JSONObject();
        p.put("read", directory.canRead());
        p.put("write", directory.canWrite());
        p.put("execute", directory.canRead());
        return p;
    }

    interface CloudStorageService {
        void uploadFile(File file, CloudCallback callback);
        void downloadFile(String cloudId, CloudCallback callback);
        boolean checkSyncStatus(File file);
    }

    interface CloudCallback {
        void onSuccess(File file);
        void onError(Exception e);
    }

    private static class CloudStorageServiceImpl implements CloudStorageService {
        @Override
        public void uploadFile(File file, CloudCallback callback) {
            new Thread(() -> {
                try {
                    Thread.sleep(1000);
                    callback.onSuccess(file);
                } catch (Exception e) { callback.onError(e); }
            }).start();
        }

        @Override
        public void downloadFile(String cloudId, CloudCallback callback) {
            new Thread(() -> {
                try {
                    Thread.sleep(1000);
                    callback.onSuccess(new File(cloudId));
                } catch (Exception e) { callback.onError(e); }
            }).start();
        }

        @Override
        public boolean checkSyncStatus(File file) {
            return file != null && file.exists() && file.length() > 0;
        }
    }
}
