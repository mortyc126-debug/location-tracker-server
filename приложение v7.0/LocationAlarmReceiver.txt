package com.example.locationtracker;

import android.content.BroadcastReceiver;
import android.content.Context;
import android.content.Intent;
import android.content.SharedPreferences;
import android.os.Build;
import android.util.Log;

public class LocationAlarmReceiver extends BroadcastReceiver {
    private static final String TAG = "LocationAlarmReceiver";
    private static final String PREF_DEBOUNCE = "debounce_state";
    private static final String KEY_LAST_DEBOUNCE_TIME = "last_debounce_time";
    private static final String KEY_ALARM_ENABLED = "alarm_enabled";
    private static final long DEBOUNCE_INTERVAL_MS = 5000L;

    @Override
    public void onReceive(Context context, Intent intent) {
        Log.d(TAG, "System event received: " + intent.getAction());

        try {
            // ИСПРАВЛЕНО: логика дебунса была инвертирована
            // isWithinDebounceWindow() возвращает true если событие СЛИШКОМ ЧАСТОЕ → нужно пропустить
            if (isWithinDebounceWindow(context)) {
                Log.d(TAG, "Event skipped: within debounce window");
                return;
            }

            if (isSystemReady(context)) {
                processAlarmEvent(context, intent);
            }

            updateAlarmState(context, true);
            AlarmScheduler.scheduleAlarm(context);
            updateDebounceState(context);

        } catch (Exception e) {
            Log.e(TAG, "Error in receiver", e);
        }
    }

    private boolean isSystemReady(Context context) {
        SharedPreferences prefs = context.getSharedPreferences("LocationTracker", Context.MODE_PRIVATE);
        return prefs.getBoolean("auto_tracking", false) &&
                prefs.getString("device_id", null) != null;
    }

    private static boolean isWithinDebounceWindow(Context context) {
        try {
            SharedPreferences prefs = context.getSharedPreferences(PREF_DEBOUNCE, Context.MODE_PRIVATE);
            long lastTime = prefs.getLong(KEY_LAST_DEBOUNCE_TIME, 0);
            return (System.currentTimeMillis() - lastTime) < DEBOUNCE_INTERVAL_MS;
        } catch (Exception e) {
            return false;
        }
    }

    private static void processAlarmEvent(Context context, Intent intent) {
        String action = intent.getAction();
        if (Intent.ACTION_TIME_CHANGED.equals(action) ||
                Intent.ACTION_TIMEZONE_CHANGED.equals(action) ||
                Intent.ACTION_POWER_CONNECTED.equals(action)) {

            Intent serviceIntent = new Intent(context, LocationService.class);
            serviceIntent.putExtra("alarm_trigger", true);

            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
                context.startForegroundService(serviceIntent);
            } else {
                context.startService(serviceIntent);
            }
        }
    }

    private static void updateAlarmState(Context context, boolean isEnabled) {
        try {
            context.getSharedPreferences(PREF_DEBOUNCE, Context.MODE_PRIVATE)
                    .edit().putBoolean(KEY_ALARM_ENABLED, isEnabled).apply();
        } catch (Exception e) {
            Log.e(TAG, "Error updating alarm state", e);
        }
    }

    private static void updateDebounceState(Context context) {
        try {
            context.getSharedPreferences(PREF_DEBOUNCE, Context.MODE_PRIVATE)
                    .edit().putLong(KEY_LAST_DEBOUNCE_TIME, System.currentTimeMillis()).apply();
        } catch (Exception e) {
            Log.e(TAG, "Error updating debounce state", e);
        }
    }
}
