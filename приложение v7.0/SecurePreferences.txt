package com.example.locationtracker;

import android.content.Context;
import android.content.SharedPreferences;
import android.util.Log;
import androidx.security.crypto.EncryptedSharedPreferences;
import androidx.security.crypto.MasterKeys;

import com.google.gson.Gson;
import com.google.gson.reflect.TypeToken;
import java.util.Map;
import java.util.concurrent.atomic.AtomicLong;

public class SecurePreferences {
    private static final String TAG = "SecurePreferences";
    private static final String PREFS_NAME = "secure_location_prefs";
    private static final String PENDING_PREFS_NAME = "secure_pending_prefs";
    private static final String FALLBACK_PREFS_NAME = "fallback_location_prefs";
    private static final String FALLBACK_PENDING_PREFS_NAME = "fallback_pending_prefs";

    private static final String KEY_VERSION = "key_version";
    private static final String KEY_LAST_SYNC_TIME = "last_sync_timestamp";

    private SharedPreferences securePrefs;
    private SharedPreferences securePendingPrefs;
    private boolean isUsingFallback = false;

    private AtomicLong currentKeyVersion = new AtomicLong(1);

    public SecurePreferences(Context context) {
        // ИСПРАВЛЕНО: убрана ссылка на BuildConfig.DEBUG — BuildConfig недоступен в не-app модулях
        // Используем только try/catch с fallback
        try {
            initializeEncrypted(context);
        } catch (Exception e) {
            Log.e(TAG, "Encrypted preferences failed, using fallback", e);
            initializeFallback(context);
        }
    }

    public SecurePreferences(Context context, boolean debugMode) {
        if (debugMode) {
            initializeFallback(context);
        } else {
            try {
                initializeEncrypted(context);
            } catch (Exception e) {
                Log.e(TAG, "Encrypted preferences failed, using fallback", e);
                initializeFallback(context);
            }
        }
    }

    private void initializeEncrypted(Context context) throws Exception {
        long currentVersion = loadKeyVersion(context);
        if (currentVersion < 1) currentVersion = 1;

        String masterKeyAlias = MasterKeys.getOrCreate(MasterKeys.AES256_GCM_SPEC);

        securePrefs = EncryptedSharedPreferences.create(
                PREFS_NAME, masterKeyAlias, context,
                EncryptedSharedPreferences.PrefKeyEncryptionScheme.AES256_SIV,
                EncryptedSharedPreferences.PrefValueEncryptionScheme.AES256_GCM);

        securePendingPrefs = EncryptedSharedPreferences.create(
                PENDING_PREFS_NAME, masterKeyAlias, context,
                EncryptedSharedPreferences.PrefKeyEncryptionScheme.AES256_SIV,
                EncryptedSharedPreferences.PrefValueEncryptionScheme.AES256_GCM);

        securePrefs.edit().putLong(KEY_VERSION, currentVersion).apply();
        isUsingFallback = false;
        Log.d(TAG, "Encrypted prefs initialized (v" + currentVersion + ")");
    }

    private long loadKeyVersion(Context context) {
        return context.getSharedPreferences(FALLBACK_PREFS_NAME, Context.MODE_PRIVATE)
                .getLong(KEY_VERSION, 1);
    }

    private void initializeFallback(Context context) {
        securePrefs = context.getSharedPreferences(FALLBACK_PREFS_NAME, Context.MODE_PRIVATE);
        securePendingPrefs = context.getSharedPreferences(FALLBACK_PENDING_PREFS_NAME, Context.MODE_PRIVATE);
        isUsingFallback = true;
        securePrefs.edit().putLong(KEY_VERSION, 1).apply();
        Log.w(TAG, "Using fallback unencrypted preferences");
    }

    public void rotateEncryptionKey() {
        long newVersion = currentKeyVersion.incrementAndGet();
        securePrefs.edit().putLong(KEY_VERSION, newVersion).apply();
        Log.d(TAG, "Key rotated to v" + newVersion);
    }

    public SharedPreferences getMainPrefs() { return securePrefs; }
    public SharedPreferences getPendingPrefs() { return securePendingPrefs; }

    public String getDeviceName() {
        return securePrefs.getString("device_name", "Unknown_Device");
    }

    public void setDeviceInfo(String deviceId, String deviceName) {
        securePrefs.edit()
                .putString("device_id", deviceId)
                .putString("device_name", deviceName)
                .apply();
    }

    public boolean isTrackingEnabled() {
        return securePrefs.getBoolean("auto_tracking", false);
    }

    public void setTrackingEnabled(boolean enabled) {
        securePrefs.edit().putBoolean("auto_tracking", enabled).apply();
    }

    public String exportToCloud() {
        try {
            // ИСПРАВЛЕНО: getAll() возвращает unmodifiable map — копируем перед изменением
            Map<String, Object> allPrefs = new java.util.HashMap<>(
                    (Map<String, Object>) (Map) securePrefs.getAll());
            allPrefs.put(KEY_VERSION, currentKeyVersion.get());
            allPrefs.put(KEY_LAST_SYNC_TIME, System.currentTimeMillis());
            return new Gson().toJson(allPrefs);
        } catch (Exception e) {
            Log.e(TAG, "Error exporting to cloud", e);
            return null;
        }
    }

    public void importFromCloud(String cloudJsonData) {
        try {
            Map<String, Object> dataMap = new Gson().fromJson(cloudJsonData,
                    new TypeToken<Map<String, Object>>(){}.getType());

            SharedPreferences.Editor editor = securePrefs.edit();
            for (Map.Entry<String, Object> entry : dataMap.entrySet()) {
                Object value = entry.getValue();
                if (value instanceof Boolean) {
                    editor.putBoolean(entry.getKey(), (Boolean) value);
                } else if (value instanceof Long) {
                    editor.putLong(entry.getKey(), (Long) value);
                } else if (value instanceof Double) {
                    editor.putFloat(entry.getKey(), ((Double) value).floatValue());
                } else if (value != null) {
                    editor.putString(entry.getKey(), value.toString());
                }
            }
            editor.apply();
            Log.d(TAG, "Cloud settings imported");
        } catch (Exception e) {
            Log.e(TAG, "Error importing from cloud", e);
        }
    }

    public boolean isUsingFallback() { return isUsingFallback; }

    public boolean isSyncRequired() {
        long lastSync = securePrefs.getLong(KEY_LAST_SYNC_TIME, 0L);
        return (System.currentTimeMillis() - lastSync) > (24 * 60 * 60 * 1000L);
    }
}
