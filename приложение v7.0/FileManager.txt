package com.example.locationtracker;

import android.content.Context;
import android.content.Intent;
import android.content.IntentFilter; // ИСПРАВЛЕНО: добавлен импорт
import android.net.Uri;
import android.os.Environment;
import android.util.Log;
import java.io.File;
import java.io.FileFilter;
import java.util.*;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.ExecutorService;  // ИСПРАВЛЕНО: добавлен импорт
import java.util.concurrent.Executors;         // ИСПРАВЛЕНО: добавлен импорт

public class FileManager {
    private static final String TAG = "FileManager";

    private final Map<String, FileMetadata> directoryCache = new ConcurrentHashMap<>();
    private final ExecutorService scannerExecutor;
    // ИСПРАВЛЕНО: FileObserver(Context) не существует в Android API
    // android.os.FileObserver принимает путь (String), не Context
    // Переименован в fileWatcher чтобы не конфликтовал с android.os.FileObserver
    private final SearchCriteria searchCriteria;
    private final Context appContext; // ИСПРАВЛЕНО: сохраняем контекст для использования в методах

    public FileManager(Context context) {
        this.appContext = context.getApplicationContext();
        this.scannerExecutor = Executors.newCachedThreadPool();
        this.searchCriteria = new SearchCriteria();
        registerStorageMonitor(context);
        Log.d(TAG, "FileManager initialized");
    }

    public void scanDirectory(File rootDirectory, int maxDepth) {
        scannerExecutor.execute(() -> {
            try {
                Log.d(TAG, "Starting scan: " + rootDirectory.getAbsolutePath());
                scanRecursive(rootDirectory, maxDepth, new FileHandler() {
                    @Override
                    public void onFileFound(File file) {
                        updateFileMetadata(file);
                    }
                    @Override
                    public void onDirectoryFound(File directory) {
                        updateDirectoryMetadata(directory);
                    }
                });
                Log.d(TAG, "Scan completed");
            } catch (Exception e) {
                Log.e(TAG, "Error during scan", e);
            }
        });
    }

    private void scanRecursive(File directory, int currentDepth, FileHandler handler) {
        if (currentDepth <= 0) return;
        File[] files = directory.listFiles();
        if (files != null) {
            for (File file : files) {
                if (file.isDirectory()) {
                    handler.onDirectoryFound(file);
                    scanRecursive(file, currentDepth - 1, handler);
                } else {
                    handler.onFileFound(file);
                }
            }
        }
    }

    private void updateFileMetadata(File file) {
        FileMetadata metadata = new FileMetadata(
                file.getAbsolutePath(),
                file.length(),
                file.lastModified(),
                getFileType(file.getName())
        );
        directoryCache.put(file.getAbsolutePath(), metadata);
    }

    // ИСПРАВЛЕНО: добавлен недостающий метод updateDirectoryMetadata
    private void updateDirectoryMetadata(File directory) {
        FileMetadata metadata = new FileMetadata(
                directory.getAbsolutePath(),
                0,
                directory.lastModified(),
                "Directory"
        );
        directoryCache.put(directory.getAbsolutePath(), metadata);
    }

    private String getFileType(String fileName) {
        switch (getFileExtension(fileName)) {
            case "jpg": case "png": case "gif": return "Image";
            case "mp4": case "avi": case "mov": return "Video";
            case "mp3": case "wav": return "Audio";
            case "pdf": case "doc": case "txt": return "Document";
            default: return "Other";
        }
    }

    private String getFileExtension(String fileName) {
        int i = fileName.lastIndexOf('.');
        return (i > 0) ? fileName.substring(i + 1).toLowerCase() : "";
    }

    public List<FileMetadata> searchFiles(FileFilter filter) {
        List<FileMetadata> results = new ArrayList<>();
        for (Map.Entry<String, FileMetadata> entry : directoryCache.entrySet()) {
            if (filter == null || filter.accept(new File(entry.getKey()))) {
                results.add(entry.getValue());
            }
        }
        return results;
    }

    private void registerStorageMonitor(Context context) {
        IntentFilter intentFilter = new IntentFilter();
        intentFilter.addAction(Intent.ACTION_MEDIA_MOUNTED);
        intentFilter.addAction(Intent.ACTION_MEDIA_REMOVED);
        intentFilter.addAction(Intent.ACTION_MEDIA_EJECT);
        context.registerReceiver(new StorageMonitorReceiver(), intentFilter);
    }

    private static class StorageMonitorReceiver extends android.content.BroadcastReceiver {
        @Override
        public void onReceive(Context context, Intent intent) {
            // ИСПРАВЛЕНО: убран каст context instanceof FileManager
            // Context никогда не является FileManager
            if (intent.getAction() != null) {
                Log.d("StorageMonitor", "Storage event: " + intent.getAction());
            }
        }
    }

    public interface FileHandler {
        void onFileFound(File file);
        void onDirectoryFound(File directory);
    }

    public static class FileMetadata {
        public final String path;
        public final long size;
        public final long lastModified;
        public final String type;
        public final Map<String, Object> extraData;

        public FileMetadata(String path, long size, long lastModified, String type) {
            this.path = path;
            this.size = size;
            this.lastModified = lastModified;
            this.type = type;
            this.extraData = new HashMap<>();
        }
    }

    public static class SearchCriteria {
        private long minSize = 0;
        private long maxSize = Long.MAX_VALUE;
        private List<String> allowedTypes;

        public SearchCriteria() {
            allowedTypes = new ArrayList<>();
        }

        public void addType(String type) { allowedTypes.add(type); }
        public boolean matchesType(String fileType) {
            return allowedTypes.isEmpty() || allowedTypes.contains(fileType);
        }
    }
}
